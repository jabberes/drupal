%{auth_method, internal}.

{auth_method, ldap}.
{ldap_servers, ["nodo.ohc.cu"]}. % List of LDAP servers
{ldap_port, 389}.
{ldap_base, "DC=nodo,DC=ohc,DC=cu"}. % Search base of LDAP directory
{ldap_rootdn, "CN=Administrator,CN=Users,DC=nodo,DC=ohc,DC=cu"}. % LDAP manager
{ldap_password, "**********"}. % Password to LDAP manager
{ldap_uidattr, "sAMAccountName"}.
{ldap_filter, "(memberOf=CN=Users,DC=nodo,DC=ohc,DC=cu)"}.

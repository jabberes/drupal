Éste es mi config.php:

// Enter your jabber server and port; if you don't have one,
// visit http://www.jabber.org/user/publicservers.php for a
// list of public servers.
define("JABBER_SERVER","jabber.org");
define("JABBER_PORT","5222");

// Enter the domain names for each of your transports; this
// is used to determine which IM network a contact is using.
// (eg: if you enter "msn.yourdomain.com" for your MSN
// transport, and a contact's Jabber ID is "john%hotmail.com@msn.yourdomain.com",
// WebMessenger will know that this is an MSN user)
define("TRANSPORT_MSN","msn.yourdomain.com");
define("TRANSPORT_ICQ","icq.jabber.org");
define("TRANSPORT_AIM","");
define("TRANSPORT_YIM","");

// Enter the commands used to determine whether or not your
// Jabber server and individual transports are running.
//
// It doesn't matter what commands you use, nor how they
// determine your Jabber server/transport status, but
// they MUST return one or more characters when running,
// and 0 characters when offline. If WebMessenger is on
// the same machine as your Jabber server, it's usually
// sufficient to just check the process list, as shown
// below.
define("JABBER_RUNNING","ps ax | grep jabberd");
define("MSN_RUNNING","ps ax | grep msntrans");
define("ICQ_RUNNING","ps ax | grep icqv7ext");
define("YIM_RUNNING","");
define("AIM_RUNNING","");

// Enter the Jabber resource used by WebMessenger. This
// can be basically anything.
define("JABBER_RESOURCE","WebMsgr");

// Enter the default font face(s)/size used by WebMessenger.
$font_family = "Tahoma,Verdana,Arial,Helvetica,sans-serif";
$font_size = 8;

// Enter the theme you're using for WebMessenger. Probably
// just "default" for now, as theme support is not yet complete.
$theme = "default";

// Enter the language to be used by WebMessenger. For English,
// this should just be "english". For other languages, you should
// make sure your country code exists in the "locale" subfolder,
// then enter the country code below (eg: de_DE for Deutsch, etc.)
//
// Note: If your language doesn't exist, and you care to translate
// WebMessenger to your language, please submit your lang_xxx.php
// file for inclusion in the next version of WebMessenger!
define("DEFAULT_LANGUAGE","english");

// If true, users will be allowed to sign up with their existing
// Jabber accounts. If false, WebMessenger will always create
// a new Jabber account for the user.
define("ALLOW_EXISTING_ACCOUNTS",true);

// Debugging mode; should usually be set to 0
define("DEBUG_MODE",0);
?>
